API Usage Examples
Python
import requests

headers = {
    'X-API-Key': 'YOUR_API_KEY',
    'Content-Type': 'application/json'
}

data = {
    'tool_id': 'portscan',
    'parameters': {
        'target': 'example.com',
        'ports': '1-1000'
    }
}

response = requests.post(
    'http://localhost:5000/api/v1/scan',
    headers=headers,
    json=data
)
print(response.json())
cURL
curl -X POST http://localhost:5000/api/v1/scan \
  -H "X-API-Key: YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "tool_id": "portscan",
    "parameters": {
      "target": "example.com",
      "ports": "1-1000"
    }
  }'
